async function submitToGithub() { const token = document.getElementById('gh-token').value.trim(); // Boşlukları temizler const title = document.getElementById('h-title').value; const desc = document.getElementById('h-desc').value; const status = document.getElementById('status'); // --- BURAYI KENDİ BİLGİLERİNE GÖRE DOLDUR --- // GitHub'da deponun en üstünde yazan ismi tam buraya yaz: const repo = "seftali-stüdyosu/seftali-stüdyosu"; const path = "data.json"; // -------------------------------------------- if(!token || !title || !desc) { alert("Lütfen anahtarı ve haber alanlarını doldurun!"); return; } status.innerText = "Bağlantı kuruluyor..."; status.style.color = "#888"; try { // 1. Dosyayı kontrol et const res = await fetch(`https://api.github.com/repos/${repo}/contents/${path}`, { headers: { "Authorization": `token ${token}` } }); if (res.status === 404) { throw new Error("data.json dosyası depoda bulunamadı! Adını kontrol et."); } else if (res.status === 401) { throw new Error("Token (Anahtar) hatalı veya süresi dolmuş."); } else if (!res.ok) { throw new Error("GitHub bağlantı hatası: " + res.status); } const data = await res.json(); const content = JSON.parse(atob(data.content)); // 2. Yeni veriyi ekle content.haberler.unshift({ id: Date.now(), baslik: title, tarih: new Date().toLocaleDateString('tr-TR'), detay: desc }); // 3. GitHub'a geri gönder const update = await fetch(`https://api.github.com/repos/${repo}/contents/${path}`, { method: "PUT", headers: { "Authorization": `token ${token}`, "Content-Type": "application/json" }, body: JSON.stringify({ message: "Panel: Haber eklendi", content: btoa(unescape(encodeURIComponent(JSON.stringify(content, null, 2)))), sha: data.sha }) }); if(update.ok) { status.style.color = "#00ff00"; status.innerText = "BAŞARILI! Haber eklendi."; document.getElementById('h-title').value = ""; document.getElementById('h-desc').value = ""; } else { throw new Error("Güncelleme sırasında hata oluştu."); } } catch (e) { status.style.color = "#ff4444"; status.innerText = "HATA: " + e.message; } }
